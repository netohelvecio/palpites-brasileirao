**Goal:** Estender o port `WhatsAppClient` com `sendPollToGroup`, implementar nas 3 impls (Baileys/stub/disabled), criar 3 templates (`tie_poll`, `tie_emoji_fallback`, `admin_picked`), e adicionar `notifyTieCandidatesPoll` no `WhatsAppNotifier` com fallback automático texto+emoji.

```ts
// apps/api/app/integrations/whatsapp/templates/tie_poll.ts
export interface TiePollCandidate {
  homeTeam: string
  awayTeam: string
  position: number
}

export interface TiePollInput {
  roundNumber: number
  candidates: TiePollCandidate[]
}

export interface TiePollOutput {
  question: string
  options: string[]
}

export function tiePollMessage(input: TiePollInput): TiePollOutput {
  const sorted = [...input.candidates].sort((a, b) => a.position - b.position)
  return {
    question: `🗳️ Empate na escolha do jogo da Rodada ${input.roundNumber} — vote no jogo da rodada!`,
    options: sorted.map((c) => `${c.homeTeam} x ${c.awayTeam}`),
  }
}
```

```ts
// apps/api/app/integrations/whatsapp/templates/tie_emoji_fallback.ts
export interface TieEmojiFallbackCandidate {
  homeTeam: string
  awayTeam: string
  position: number
}

export interface TieEmojiFallbackInput {
  roundNumber: number
  candidates: TieEmojiFallbackCandidate[]
}

const NUMBER_EMOJIS = [
  '1️⃣',
  '2️⃣',
  '3️⃣',
  '4️⃣',
  '5️⃣',
  '6️⃣',
  '7️⃣',
  '8️⃣',
  '9️⃣',
  '🔟',
]

export function tieEmojiFallbackMessage(input: TieEmojiFallbackInput): string {
  const sorted = [...input.candidates].sort((a, b) => a.position - b.position)
  const lines = sorted.map((c, idx) => {
    const emoji = NUMBER_EMOJIS[idx] ?? `${idx + 1}.`
    return `${emoji} ${c.homeTeam} x ${c.awayTeam}`
  })
  return [
    `🗳️ *Empate na escolha do jogo da Rodada ${input.roundNumber}!*`,
    '',
    'Vote reagindo com o número correspondente:',
    '',
    ...lines,
    '',
    '(o admin homologa a escolha mais votada)',
  ].join('\n')
}
```

```ts
  async sendPollToGroup(question: string, options: string[]): Promise<{ messageId: string }> {
    const jid = env.get('WHATSAPP_GROUP_JID')
    if (!jid) {
      throw new Error('WHATSAPP_GROUP_JID não configurado')
    }
    if (!this.socket || this.state !== 'open') {
      throw new Error('BaileysClient: socket não conectado')
    }
    const result = await this.socket.sendMessage(jid, {
      poll: { name: question, values: options, selectableCount: 1 },
    })
    return { messageId: result?.key?.id ?? '' }
  }
```

```ts
// apps/api/tests/unit/whatsapp/notifier_tie_poll.spec.ts
import { test } from '@japa/runner'
import WhatsAppNotifier from '#services/whatsapp_notifier'
import WhatsAppClient, {
  type IncomingMessageHandler,
  type PollSendResult,
  type WhatsAppMode,
} from '#integrations/whatsapp/whatsapp_client'

class FakeClient extends WhatsAppClient {
  readonly mode: WhatsAppMode = 'stub'
  pollCalls: { question: string; options: string[] }[] = []
  groupCalls: string[] = []
  pollImpl: (q: string, o: string[]) => Promise<PollSendResult> = async (q, o) => {
    this.pollCalls.push({ question: q, options: o })
    return { messageId: 'msg-1' }
  }

  async connect() {}
  async disconnect() {}
  isConnected() {
    return true
  }
  async sendToGroup(text: string) {
    this.groupCalls.push(text)
  }
  async sendToUser() {}
  async sendPollToGroup(question: string, options: string[]) {
    return this.pollImpl(question, options)
  }
  onMessage(_h: IncomingMessageHandler) {}
}

const PAYLOAD = {
  roundNumber: 12,
  candidates: [
    { homeTeam: 'A', awayTeam: 'B', position: 1 },
    { homeTeam: 'C', awayTeam: 'D', position: 2 },
  ],
}

test.group('WhatsAppNotifier.notifyTieCandidatesPoll', () => {
  test('poll OK → mode=poll, messageId retornado', async ({ assert }) => {
    const fake = new FakeClient()
    const notifier = new WhatsAppNotifier(fake)
    const r = await notifier.notifyTieCandidatesPoll(PAYLOAD)
    assert.equal(r.mode, 'poll')
    assert.equal(r.messageId, 'msg-1')
    assert.lengthOf(fake.pollCalls, 1)
    assert.lengthOf(fake.groupCalls, 0)
  })

  test('poll throws → fallback emoji, mode=emoji, messageId=null', async ({ assert }) => {
    const fake = new FakeClient()
    fake.pollImpl = async () => {
      throw new Error('poll not supported')
    }
    const notifier = new WhatsAppNotifier(fake)
    const r = await notifier.notifyTieCandidatesPoll(PAYLOAD)
    assert.equal(r.mode, 'emoji')
    assert.isNull(r.messageId)
    assert.lengthOf(fake.groupCalls, 1)
    assert.match(fake.groupCalls[0], /1️⃣ A x B/)
    assert.match(fake.groupCalls[0], /2️⃣ C x D/)
  })
})
```

```ts
  async notifyTieCandidatesPoll(payload: {
    roundNumber: number
    candidates: TiePollCandidate[]
  }): Promise<{ mode: 'poll' | 'emoji'; messageId: string | null }> {
    const { question, options } = tiePollMessage(payload)
    try {
      const r = await this.client.sendPollToGroup(question, options)
      return { mode: 'poll', messageId: r.messageId }
    } catch (err) {
      logger.warn({ err }, 'WhatsAppNotifier: poll falhou, fallback emoji')
      const fallback = tieEmojiFallbackMessage(payload)
      await this.client.sendToGroup(fallback)
      return { mode: 'emoji', messageId: null }
    }
  }
```
