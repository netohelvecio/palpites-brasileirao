```ts
import { MatchStatus } from '@palpites/shared'
import type { ApiFootballFixtureStatus } from './types.js'

const SCHEDULED = new Set<ApiFootballFixtureStatus>([
  'TBD', 'NS', 'PST', 'CANC', 'ABD', 'AWD', 'WO',
])
const LIVE = new Set<ApiFootballFixtureStatus>([
  '1H', 'HT', '2H', 'ET', 'BT', 'P', 'SUSP', 'INT', 'LIVE',
])
const FINISHED = new Set<ApiFootballFixtureStatus>(['FT', 'AET', 'PEN'])

export function mapApiFootballStatus(status: ApiFootballFixtureStatus): MatchStatus {
  if (FINISHED.has(status)) return MatchStatus.FINISHED
  if (LIVE.has(status)) return MatchStatus.LIVE
  if (SCHEDULED.has(status)) return MatchStatus.SCHEDULED
  // Fallback defensivo: novos status desconhecidos caem em scheduled
  return MatchStatus.SCHEDULED
}
```

- [ ] **Step 1: Escrever service**

Arquivo: `apps/api/app/services/score_refresh_service.ts`
```ts
import { inject } from '@adonisjs/core'
import ApiFootballClient from '#integrations/api_football/client'
import { mapApiFootballStatus } from '#integrations/api_football/status_mapper'
import MatchRepository from '#repositories/match_repository'
import type Match from '#models/match'

export type RefreshResult =
  | { ok: true; match: Match }
  | { ok: false; reason: string }

@inject()
export default class ScoreRefreshService {
  constructor(
    private client: ApiFootballClient,
    private matchRepository: MatchRepository
  ) {}

  async refreshByRoundId(roundId: string): Promise<RefreshResult> {
    const match = await this.matchRepository.findByRoundId(roundId)
    if (!match) return { ok: false, reason: 'round não tem jogo cadastrado' }

    const fixture = await this.client.fetchFixtureById(match.externalId)
    if (!fixture) return { ok: false, reason: 'fixture não encontrada na API-Football' }

    const status = mapApiFootballStatus(fixture.fixture.status.short)
    await this.matchRepository.update(match, {
      homeScore: fixture.goals.home,
      awayScore: fixture.goals.away,
      status,
    })
    return { ok: true, match }
  }
}
```

## Resumo do Plano 2.4

Ao final do grupo 2.x:
- `POST /seasons/:id/sync` e `POST /rounds/:roundId/match/refresh-score` totalmente funcionais.
- Mapping de status da API coberto por unit tests.
- Mock infra reutilizável em `tests/helpers/api_football_mock.ts`.
- Pronto para o Plano 3 (scheduler) chamar essas services em cron jobs.

**Decisões:**
- Refresh aceita `roundId` (não `matchId`) pra casar com o endpoint já existente e a convenção de "1 match por round".
- 422 quando round não tem match ou fixture sumiu — diferente de 404 porque o round existe.
- Fallback defensivo no status mapper: status desconhecido → `scheduled` (melhor do que crashar).
- Refresh não lida com "jogo cancelado" — status ABD/CANC mapeia pra scheduled, mas o admin pode corrigir via PUT /match se precisar. Pode ser refinado depois.
