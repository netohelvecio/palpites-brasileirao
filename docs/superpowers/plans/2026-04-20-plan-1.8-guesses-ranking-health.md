import vine from '@vinejs/vine'

export const createGuessValidator = vine.compile(
  vine.object({
    userId: vine.string().uuid(),
    matchId: vine.string().uuid(),
    homeScore: vine.number().min(0).max(20),
    awayScore: vine.number().min(0).max(20),
  })
)

export const updateGuessValidator = vine.compile(
  vine.object({
    homeScore: vine.number().min(0).max(20).optional(),
    awayScore: vine.number().min(0).max(20).optional(),
  })
)

import type { HttpContext } from '@adonisjs/core/http'
import Guess from '#models/guess'
import Match from '#models/match'
import { createGuessValidator, updateGuessValidator } from '#validators/guess_validator'

export default class GuessesController {
  async store({ request, response }: HttpContext) {
    const payload = await request.validateUsing(createGuessValidator)
    const existing = await Guess.query()
      .where('user_id', payload.userId)
      .where('match_id', payload.matchId)
      .first()
    if (existing) {
      return response.unprocessableEntity({
        error: 'palpite já existe para esse usuário e jogo',
      })
    }
    const guess = await Guess.create(payload as any)
    return response.created(guess)
  }

  async indexByRound({ params, response }: HttpContext) {
    const match = await Match.query().where('round_id', params.roundId).first()
    if (!match) return response.ok([])
    const guesses = await Guess.query().where('match_id', match.id).preload('user')
    return response.ok(guesses)
  }

  async update({ params, request, response }: HttpContext) {
    const guess = await Guess.findOrFail(params.id)
    const payload = await request.validateUsing(updateGuessValidator)
    guess.merge(payload)
    await guess.save()
    return response.ok(guess)
  }

  async destroy({ params, response }: HttpContext) {
    const guess = await Guess.findOrFail(params.id)
    guess.isDeleted = true
    await guess.save()
    return response.noContent()
  }
}
```

import type { HttpContext } from '@adonisjs/core/http'
import Score from '#models/score'
import Match from '#models/match'
import Guess from '#models/guess'
import { sortRanking, type RankingEntry } from '#services/ranking_service'

export default class RankingController {
  async bySeason({ params, response }: HttpContext) {
    const scores = await Score.query()
      .where('season_id', params.seasonId)
      .preload('user')
    const entries: RankingEntry[] = scores.map((s) => ({
      userId: s.userId,
      name: s.user.name,
      emoji: s.user.emoji,
      totalPoints: s.totalPoints,
      exactScoresCount: s.exactScoresCount,
    }))
    return response.ok(sortRanking(entries))
  }

  async byRound({ params, response }: HttpContext) {
    const match = await Match.query().where('round_id', params.roundId).first()
    if (!match) return response.ok([])
    const guesses = await Guess.query().where('match_id', match.id).preload('user')
    const rows = guesses
      .map((g) => ({
        userId: g.userId,
        name: g.user.name,
        emoji: g.user.emoji,
        points: g.points ?? 0,
      }))
      .sort((a, b) => b.points - a.points)
    return response.ok(rows)
  }
}
```
