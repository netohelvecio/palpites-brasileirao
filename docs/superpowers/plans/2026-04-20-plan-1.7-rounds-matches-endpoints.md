```ts
import type { HttpContext } from '@adonisjs/core/http'
import Round from '#models/round'
import { updateRoundStatusValidator } from '#validators/round_validator'

export default class RoundsController {
  async indexBySeason({ params, response }: HttpContext) {
    const rounds = await Round.query()
      .where('season_id', params.seasonId)
      .orderBy('number', 'asc')
    return response.ok(rounds)
  }

  async show({ params, response }: HttpContext) {
    const round = await Round.query()
      .where('id', params.id)
      .preload('match', (m) => m.preload('guesses'))
      .firstOrFail()
    return response.ok(round)
  }

  async updateStatus({ params, request, response }: HttpContext) {
    const round = await Round.findOrFail(params.id)
    const { status } = await request.validateUsing(updateRoundStatusValidator)
    round.status = status
    await round.save()
    return response.ok(round)
  }
}
```

```ts
import type { HttpContext } from '@adonisjs/core/http'
import Match from '#models/match'
import Round from '#models/round'
import { upsertMatchValidator } from '#validators/match_validator'

export default class MatchesController {
  async show({ params, response }: HttpContext) {
    const match = await Match.query().where('round_id', params.roundId).first()
    if (!match) return response.notFound({ error: 'match not found' })
    return response.ok(match)
  }

  async upsert({ params, request, response }: HttpContext) {
    const round = await Round.findOrFail(params.roundId)
    const payload = await request.validateUsing(upsertMatchValidator)

    const existing = await Match.query().where('round_id', round.id).first()
    if (existing) {
      existing.merge({
        ...payload,
        status: 'scheduled',
        homeScore: null,
        awayScore: null,
      } as any)
      await existing.save()
      return response.ok(existing)
    }

    const match = await Match.create({
      roundId: round.id,
      ...payload,
      status: 'scheduled',
    } as any)
    return response.ok(match)
  }

  async refreshScore({ params, response }: HttpContext) {
    // Stub: implementação real da API-Football vem no Plano 2.
    return response.accepted({
      message: 'refresh será implementado no plano 2',
      roundId: params.roundId,
    })
  }
}
```
